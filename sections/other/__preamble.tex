% Save original headers
\let\ams@oddhead\@oddhead
\let\ams@evenhead\@evenhead

% Make room for two lines
\setlength{\headheight}{24pt}

\def\@oddhead{%
  \vbox{%
    % page number on its own line
    \hbox to \textwidth{\hfil\thepage}%
    \vskip 4pt
    % original header, full width, no page number
    \hbox to \textwidth{%
      \let\thepage\relax
      \ams@oddhead
    }%
  }%
}

\def\@evenhead{%
  \vbox{%
    \hbox to \textwidth{\thepage\hfil}%
    \vskip 4pt
    \hbox to \textwidth{%
      \let\thepage\relax
      \ams@evenhead
    }%
  }%
}

\makeatother
